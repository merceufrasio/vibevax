    return ads.filter(ad => {
      // Skip if ad is not set to show
      if (ad.showing.is_show === false) {
        return false;
      }
      
      // Country filtering
      if (ad.showing.countries) {
        if (ad.showing.is_showing === false && ad.showing.countries.includes(location)) {
          return false;
        }
        
        if (ad.showing.is_showing === true && !ad.showing.countries.includes(location)) {
          return false;
        }
      }
      
      // Device filtering
      if (isDesktop && ad.showing.desktop) {
        return true;
      } else if (isMobile && ad.showing.mobile && ad.showing.mobile.includes(deviceType)) {
        return true;
      }
      
      return false;
