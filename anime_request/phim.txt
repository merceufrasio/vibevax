/**
 * Popup Advertisement Manager
 * 
 * Modern, clean implementation of a popup advertisement system with:
 * - Device detection
 * - Cookie management
 * - User interaction tracking
 * - Geolocation filtering
 */

class PopupManager {
  constructor(options = {}) {
    // Configuration
    this.domain = options.domain || 'avs';
    this.site = document.currentScript?.getAttribute('site');
    
    // State
    this.popupIndex = this.getCookie('_popup_index') ? parseInt(this.getCookie('_popup_index')) : 0;
    
    // Device detection
    this.detectDevices();
    
    // Initialize on page load
    window.addEventListener('load', this.initialize.bind(this));
  }

  /**
   * Detect user device type
   */
  detectDevices() {
    const userAgent = navigator.userAgent || navigator.vendor || window.opera;
    
    // iPad Pro/Air resolution detection
    const iPadProResolutions = [
      { width: 1024, height: 1366 },
      { width: 1112, height: 834 },
      { width: 1194, height: 834 },
      { width: 820, height: 1180 }
    ];
    
    this.isIPadProOrAir = iPadProResolutions.some(resolution => 
      (window.screen.width === resolution.width && window.screen.height === resolution.height) || 
      (window.screen.width === resolution.height && window.screen.height === resolution.width)
    );
    
    this.isIOS = /iPad|iPhone|iPod/.test(userAgent) || this.isIPadProOrAir;
    this.isMobile = /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(userAgent);
    this.deviceType = this.isIOS ? 'IOS' : 'ANDROID';
  }

  /**
   * Initialize the popup system
   */
  initialize() {
    // Fetch location data to filter ads by country
    fetch('https://www.cloudflare.com/cdn-cgi/trace')
      .then(response => response.text())
      .then(data => {
        const lines = data.split('\n');
        let location;
        
        for (const line of lines) {
          const [key, value] = line.split('=');
          if (key === 'loc') {
            location = value;
            break;
          }
        }
        
        this.setCookie('_location', location, 86400);
        this.loadAdsData(location);
      })
      .catch(() => {
        // Use cached location if fetch fails
        const cachedLocation = this.getCookie('_location');
        this.loadAdsData(cachedLocation);
      });
  }

  /**
   * Load and filter advertisement data
   * @param {string} location - Country code
   */
  loadAdsData(location) {
    // Load ad configuration
    const adData = this.getAdConfiguration();
    
    if (adData.popup[this.domain] && adData.popup[this.domain].items) {
      const filteredAds = this.filterAdsByLocation(
        adData.popup[this.domain].items,
        location,
        !this.isMobile,
        this.isMobile,
        this.deviceType
      );
      
      this.setupPopups(filteredAds);
    }
  }

  /**
   * Filter ads based on location and device
   */
  filterAdsByLocation(ads, location, isDesktop, isMobile, deviceType) {
    return ads.filter(ad => {
      // Skip if ad is not set to show
      if (ad.showing.is_show === false) {
        return false;
      }
      
      // Country filtering
      if (ad.showing.countries) {
        if (ad.showing.is_showing === false && ad.showing.countries.includes(location)) {
          return false;
        }
        
        if (ad.showing.is_showing === true && !ad.showing.countries.includes(location)) {
          return false;
        }
      }
      
      // Device filtering
      if (isDesktop && ad.showing.desktop) {
        return true;
      } else if (isMobile && ad.showing.mobile && ad.showing.mobile.includes(deviceType)) {
        return true;
      }
      
      return false;
    });
  }

  /**
   * Setup and display popups
   * @param {Array} popups - Filtered popup configurations
   */
  setupPopups(popups) {
    popups.forEach(popup => {
      this.createPopupElement(popup);
      this.checkVersion(popup.id, popup.version);
    });
    
    this.showFirstPopup(popups);
  }

  /**
   * Create popup DOM elements
   * @param {Object} popup - Popup configuration
   */
  createPopupElement(popup) {
    const popupHtml = `
      <div id="${popup.id}" 
           time-click-close="${popup.timeClickClose}" 
           time-click-image-to-hide="${popup.timeClickImageToHidePopup}" 
           style="display:none;position:fixed;top:50%;left:50%;transform:translate(-50%, -50%);background-color:rgba(0, 0, 0, 0.5);width:100vw;height:100vh;z-index:99999;justify-content:center;">
        <div style="text-align:center;margin:auto;display:flex;">
          <div style="text-align:center;height:fit-content;align-content:center;width:fit-content;position:relative;">
            <a class="ads-banner" rel="nofollow" id="url-${popup.id}" target="_blank" href="${popup.urlPopUnder}">
              <img class="ads-banner" id="img-${popup.id}" style="max-width:500px" 
                   src="${this.isMobile ? popup.imageUrl.mobile : popup.imageUrl.desktop}" />
            </a>
            <div class="popup-icon-close" style="position:absolute;width:50px;right:-20px;top:-20px;height:50px;z-index:9999999999;cursor:pointer;">
              <div id="close-${popup.id}" url-close="${popup.urlPopUnderClose}" 
                   style="color:#ffffff;float:right;font-size:1.5rem;font-weight:bold;display:flex;justify-content:center;align-items:center;width:40px;height:40px;border-radius:50%;background-color:#e62117;font-size:24px;display:none;">×</div>
            </div>
          </div>
        </div>
      </div>
    `;
    
    document.body.insertAdjacentHTML('beforeend', popupHtml);
    
    // Add event listeners
    this.addPopupEventListeners(popup);
  }

  /**
   * Add event listeners to popup elements
   * @param {Object} popup - Popup configuration
   */
  addPopupEventListeners(popup) {
    const popupElement = document.getElementById(popup.id);
    const closeButton = document.getElementById(`close-${popup.id}`);
    
    // Close button event
    closeButton.addEventListener('click', () => {
      if (popupElement.innerHTML.length) {
        const clickCount = this.getCookie(`_${popup.id}_timeClickClose`) 
          ? Number(this.getCookie(`_${popup.id}_timeClickClose`)) 
          : 1;
        const maxClicks = popupElement.getAttribute('time-click-close');
        const shouldAdvance = (!popup.hidePopupWhenUserClickImage && !maxClicks) || 
                             (maxClicks && clickCount >= Number(maxClicks));
        
        // Hide popup
        popupElement.style.display = 'none';
        this.setCookie(`_${popup.id}_timeClickClose`, clickCount + 1, 86400);
        
        // Handle popup URL opening
        const closeUrl = closeButton.getAttribute('url-close');
        if (closeUrl) {
          window.open(closeUrl, '_blank');
        }
        
        // Update popup state
        if (shouldAdvance) {
          this.advancePopupState(popup);
        }
      }
    });
    
    // Banner click event (if configured)
    if (popup.hidePopupWhenUserClickImage) {
      const bannerLink = document.getElementById(`url-${popup.id}`);
      if (bannerLink) {
        bannerLink.addEventListener('click', (event) => {
          event.preventDefault();
          
          if (popupElement.innerHTML.length) {
            const clickCount = this.getCookie(`_${popup.id}_timeClickImage`) 
              ? Number(this.getCookie(`_${popup.id}_timeClickImage`)) 
              : 1;
            const maxClicks = popupElement.getAttribute('time-click-image-to-hide');
            const shouldAdvance = maxClicks && clickCount >= Number(maxClicks);
            
            // Hide popup
            popupElement.style.display = 'none';
            this.setCookie(`_${popup.id}_timeClickImage`, clickCount + 1, 86400);
            
            // Open URL
            const url = bannerLink.getAttribute('href');
            if (url) {
              window.open(url, '_blank');
            }
            
            // Update popup state
            if (shouldAdvance) {
              this.advancePopupState(popup);
            }
          }
        });
      }
    }
  }

  /**
   * Advance popup state after interaction
   * @param {Object} popup - Popup configuration
   */
  advancePopupState(popup) {
    const showCount = (this.getCookie(`_${popup.id}_timeShow`) ? 
                      Number(this.getCookie(`_${popup.id}_timeShow`)) : 0) + 1;
    const currentTime = new Date().getTime();
    
    this.setCookie(`_${popup.id}_timePrevious`, currentTime, 86400);
    this.setCookie(`_${popup.id}_timeShow`, showCount, 86400);
    
    // Reset click counters
    this.setCookie(`_${popup.id}_timeClickClose`, null, 0);
    this.setCookie(`_${popup.id}_timeClickImage`, null, 0);
    
    // Reset or advance to next popup
    if (showCount >= popup.timeDisplays.length) {
      this.resetPopupStatus(popup.id);
      this.popupIndex = this.getCookie('_popup_index') 
        ? parseInt(this.getCookie('_popup_index')) 
        : 0;
      
      // Set delay for next cycle
      this.setCookie(`_${popup.id}_timeDelay`, currentTime + popup.timeLifeCycle * 1000, 86400);
    }
    
    // Advance to next popup
    this.setCookie('_popup_index', this.popupIndex + 1, 86400);
  }

  /**
   * Show first popup in sequence
   * @param {Array} popups - Popup configurations
   */
  showFirstPopup(popups) {
    this.showPopup(this.popupIndex, popups);
  }

  /**
   * Show a specific popup
   * @param {number} index - Popup index
   * @param {Array} popups - Popup configurations
   */
  showPopup(index, popups) {
    const currentTime = new Date().getTime();
    
    // Reset index if we've shown all popups
    if (index >= popups.length) {
      this.setCookie('_popup_index', 0, 86400);
      return;
    }
    
    // Hide all popups initially
    popups.forEach(popup => {
      const element = document.getElementById(popup.id);
      if (element) {
        element.style.display = 'none';
      }
    });
    
    // Get current popup
    const popup = popups[index];
    const popupElement = document.getElementById(popup.id);
    const closeButton = popupElement.querySelector(`#close-${popup.id}`);
    const displayTimes = popup.timeDisplays.length;
    
    // Check each display time
    for (let i = 0; i < displayTimes; i++) {
      const showCount = parseInt(this.getCookie(`_${popup.id}_timeShow`)) || 0;
      const lifeCycle = popup.timeLifeCycle * 1000;
      const previousTime = parseInt(this.getCookie(`_${popup.id}_timePrevious`)) || 0;
      
      // Show this popup if it's time
      if (showCount === i || currentTime - previousTime >= lifeCycle) {
        const timeDelay = this.getCookie(`_${popup.id}_timeDelay`) === null 
          ? 0 
          : +this.getCookie(`_${popup.id}_timeDelay`) - currentTime;
        
        if (timeDelay <= 0) {
          // Initialize previous time if not set
          if (this.getCookie(`_${popup.id}_timePrevious`) === null) {
            this.setCookie(`_${popup.id}_timePrevious`, currentTime, 86400);
          }
          
          const displayTime = popup.timeDisplays[i];
          const previousTimeValue = +this.getCookie(`_${popup.id}_timePrevious`);
          
          // Schedule popup display
          setTimeout(() => {
            popupElement.style.display = 'flex';
            popupElement.style.zIndex = 99999 - displayTime;
            
            // Show close button after delay if configured
            const closeDelay = typeof popup.countdownToShowClose !== 'undefined' 
              ? popup.countdownToShowClose * 1000 
              : 0;
              
            setTimeout(() => {
              closeButton.style.display = 'block';
            }, closeDelay);
          }, previousTimeValue + displayTime * 1000 - currentTime);
          
          this.setCookie(`_${popup.id}_timeDelay`, null, 0);
        }
        break;
      }
    }
  }

  /**

  setCookie(name, value, lifetime = null) {
    if (lifetime === null || lifetime === '0') {
      return;
    }
    
    const date = new Date();
    date.setTime(date.getTime() + lifetime * 1000);
    
    document.cookie = `${name}=${escape(value)};expires=${date.toUTCString()};path=/`;
  }

   * @param {string} name - Cookie name
   * @returns {string|null} Cookie value or null if not found
   */
  getCookie(name) {
    const cookies = `; ${document.cookie}`;
    const parts = cookies.split(`; ${name}=`);
    
    if (parts.length === 2) {
      return parts.pop().split(';').shift();
    }
    
    return null;
  }
